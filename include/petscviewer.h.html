301: PETSC_STATIC_INLINE PetscErrorCode PetscViewerFlowControlStepMaster(PetscViewer viewer,PetscInt i,PetscInt *mcnt,PetscInt cnt)
302: {
304:   MPI_Comm       comm;

307:   PetscObjectGetComm((PetscObject)viewer,&comm);
308:   if (i >= *mcnt) {
309:     *mcnt += cnt;
310:     MPI_Bcast(mcnt,1,MPIU_INT,0,comm);
311:   }
312:   return(0);
313: }

317: PETSC_STATIC_INLINE PetscErrorCode PetscViewerFlowControlEndMaster(PetscViewer viewer,PetscInt *mcnt)
318: {
320:   MPI_Comm       comm;
322:   PetscObjectGetComm((PetscObject)viewer,&comm);
323:   *mcnt = 0;
324:   MPI_Bcast(mcnt,1,MPIU_INT,0,comm);
325:   return(0);
326: }

330: PETSC_STATIC_INLINE PetscErrorCode PetscViewerFlowControlStepWorker(PetscViewer viewer,PetscMPIInt rank,PetscInt *mcnt)
331: {
333:   MPI_Comm       comm;
335:   PetscObjectGetComm((PetscObject)viewer,&comm);
336:   while (PETSC_TRUE) {
337:     if (rank < *mcnt) break;
338:     MPI_Bcast(mcnt,1,MPIU_INT,0,comm);
339:   }
340:   return(0);
341: }

345: PETSC_STATIC_INLINE PetscErrorCode PetscViewerFlowControlEndWorker(PetscViewer viewer,PetscInt *mcnt)
346: {
348:   MPI_Comm       comm;
350:   PetscObjectGetComm((PetscObject)viewer,&comm);
351:   while (PETSC_TRUE) {
352:     MPI_Bcast(mcnt,1,MPIU_INT,0,comm);
353:     if (!*mcnt) break;
354:   }
355:   return(0);
356: }
